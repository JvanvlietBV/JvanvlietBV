# Directory Structure

Laravel Docsopen in new window

Added Directory App\Http\Services

We use a very simplistic class extraction to move our business logic from the controller to a service class.

These classes follow Laravel resource convention were possible, i.e. index, show,store etc. but will sometimes include custom functions.

# Validation

Laravel FormRequestopen in new windows

We validate every Store and Update Request with a FormRequest.
By default the authorize() function is set to true, because we use Middleware in our Controllers to check the role and permissions of the user.

<?php

namespace App\Http\Requests;

use Illuminate\Foundation\Http\FormRequest;

class AnalyticStoreRequest extends FormRequest
{
    /**
     * Determine if the user is authorized to make this request.
     *
     * @return bool
     */
    public function authorize()
    {
        return true;
    }

    /**
     * Get the validation rules that apply to the request.
     *
     * @return array
     */
    public function rules()
    {
        return [
            'department' => 'required',
            'sub_department' => 'required',
            'metric' => 'required',
            'location' => 'required',
            'year' => 'sometimes',
            'month' => 'sometimes',
            'week' => 'sometimes',
            'period' => 'sometimes',
            'country' => 'sometimes',
            'source' => 'required',
            'title' => 'required',
            'status' => 'required',
        ];
    }
}

# Controllers

Laravel Docsopen in new window

In our Controllers we use middleware to determine if a user has a certain role to ensure only those that are authorized to do so, can fire certain functions.

<?php

namespace App\Http\Controllers;

use App\Models\Analytic;
use Illuminate\Http\Request;

use App\Http\Requests\AnalyticStoreRequest;
use App\Http\Services\AnalyticsService;

class AnalyticController extends Controller
{

     /**
     * Middleware to Ensure that admin functions cannot be excecuted unless user is admin
     *
     * @return \Illuminate\Http\Response
     */
    public function __construct()
    {
        $this->middleware('EnsureUserIsAdmin')->except(['index', 'show']);
    }

    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index(Request $request)
    {
        return AnalyticsService::index($request);
    }

    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function admin_index(Request $request)
    {
        return AnalyticsService::admin_index($request);
    }

    /**
     * Show the form for creating a new resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function create()
    {
        return AnalyticsService::create();
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(AnalyticStoreRequest $request)
    {
        return AnalyticsService::store($request);
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Analytic  $analytic
     * @return \Illuminate\Http\Response
     */
    public function show(Analytic $analytic)
    {
        return AnalyticsService::show($analytic);
    }

    /**
     * Show the form for editing the specified resource.
     *
     * @param  \App\Models\Analytic  $analytic
     * @return \Illuminate\Http\Response
     */
    public function edit(Request $request, Analytic $analytic)
    {
        return AnalyticsService::edit($request, $analytic);
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Models\Analytic  $analytic
     * @return \Illuminate\Http\Response
     */
    public function update(AnalyticStoreRequest $request, Analytic $analytic)
    {
        return AnalyticsService::update($request, $analytic);
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Models\Analytic  $analytic
     * @return \Illuminate\Http\Response
     */
    public function destroy(Analytic $analytic)
    {
        return AnalyticsService::destroy($analytic);
    }
}

# Services

We use service classes to put our logic in.
These services will always return an Inertia::Render or Redirect response.

Note we have to declare these functions static as our controllers call them statically.

<?php

namespace App\Http\Services;

use Illuminate\Http\Request;
use Session;
use Redirect;
use Illuminate\Auth\Access\Response;
use Illuminate\Support\Facades\Storage;
use Image;
use Inertia\Inertia;


use App\Models\Analytic;

use App\Http\Requests\AnalyticStoreRequest;


class AnalyticsService
{

    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public static function index($request)
    {

        $user = auth()->user();

        $itemsPerPage = $request->filled('itemsPerPage') ? $request->itemsPerPage : 24;
        $orderBy = $request->filled('orderBy') ? $request->orderBy : 'desc';
        $sortBy = $request->filled('sortBy') ? $request->sortBy : 'created_at';

        if($user->datapermissions->contains('name','all')|| $user->isSuperuser || $user->isAdministrator){
            $default = array(
                'ecommerce',
                'potplants',
                'flowers',
                'other',
                'both'
            );
        } else {
            $default = array('both','none');
            foreach($user['datapermissions'] as $permission){
                array_push($default, $permission['name']);
            };
        }


        return Inertia::render('Analytics/Index', [
            'title' => 'Analytics',
            'results' => Analytic::orderBy($sortBy, $orderBy)
                ->where('status',1)
                ->whereIn('department', $default)
                ->when(!$request->exists('department') && $user->default_department && null == $request->query('page'), function($query) use ($request, $user) {

                    $query->where('department', $user->default_department);
                })
                ->when(!empty($request->get('department')), function($query) use ($request) {
                    $departmentArray = explode(',',$request->get('department'));
                    $query->whereIn('department', $departmentArray);
                })
                ->when(!empty($request->get('location')), function($query) use ($request) {
                    $query->where('location', $request->get('location'));
                })
                ->when(!empty($request->get('sub_department')), function($query) use ($request) {
                    $query->where('sub_department', $request->get('sub_department'));
                })
                ->when(!empty($request->get('country')), function($query) use ($request) {
                    $query->where('country', $request->get('country'));
                })
                ->when(!empty($request->get('metric')), function($query) use ($request) {
                    $query->where('metric', $request->get('metric'));
                })
                ->when(!empty($request->get('year')), function($query) use ($request) {
                $yearArray = explode(',',$request->get('year'));
                $query->whereIn('year', $yearArray);

                if(in_array(date('Y'), $yearArray)){
                    $query->orWhereNull('year');
                }
                })
                ->when(!empty($request->get('month')), function($query) use ($request) {
                    $monthArray = explode(',',$request->get('month'));
                    $query->whereIn('month', $monthArray);
                })
                ->when(!empty($request->get('period')), function($query) use ($request) {
                    $periodArray = explode(',',$request->get('period'));
                    $query->whereIn('period', $periodArray);
                })
                ->when(!empty($request->get('search')), function($query) use ($request) {
                    $query->where('title', 'like', "%{$request->search}%");
                    $query->orWhere('department', 'like', "%{$request->search}%");
                    $query->orWhere('location', 'like', "%{$request->search}%");
                    $query->orWhere('sub_department', 'like', "%{$request->search}%");
                    $query->orWhere('year', 'like', "%{$request->search}%");
                    $query->orWhere('country', 'like', "%{$request->search}%");
                    $query->orWhere('metric', 'like', "%{$request->search}%");

                })
                ->paginate($itemsPerPage)
                ->withQueryString()
                ->through(
                    fn($analytic) => [
                        'id' => $analytic->id,
                        'title' => $analytic->title,
                        'year' => $analytic->year,
                        'department' => $analytic->department,
                        'status' => $analytic->status,
                        'thumbnail' => $analytic->thumbnail,
                    ]
                ),

            'filters' => [
                'search' => $request->search,
                'location' => $request->location,
                'department' => $request->exists('department') ||null !== $request->query('page') ? $request->department : $user->default_department,
                'sub_department' => $request->sub_department,
                'metric' => $request->metric,
                'year' => $request->year,
                'month' => $request->month,
                'period' => $request->period,
                'country' => $request->country,
                'itemsPerPage' => $request->itemsPerPage,
                'orderBy' => $orderBy,
                'sortBy' => $sortBy,
            ]
        ]);
    }

    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public static function admin_index($request)
    {

        $itemsPerPage = $request->has('itemsPerPage') ? $request->itemsPerPage : 8;
        $orderBy = $request->has('orderBy') ? $request->orderBy : 'desc';
        $sort = $request->has('sortBy') ? $request->sortBy : 'created_at';

        return Inertia::render('AdminDashboard/Analytics/Index', [
            'title' => 'Analytics',
            'analytics' => Analytic::orderBy($sort, $orderBy)
                ->when(!empty($request->get('search')), function($query) use ($request) {
                    $query->where('title', 'like', "%{$request->search}%");
                    $query->orWhere('department', 'like', "%{$request->search}%");
                    $query->orWhere('sub_department', 'like', "%{$request->search}%");
                    $query->orWhere('metric', 'like', "%{$request->search}%");

                })
                ->paginate($itemsPerPage)
                ->withQueryString()
                ->through(
                    fn($item) => [
                        'id' => $item->id,
                        'title' => $item->title,
                        'status' => $item->status,
                    ]
                ),

            'filters' => [
                'search' => $request->search,
            ]
        ]);
    }



    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public static function show($analytic)
    {
        return Inertia::render('Analytics/Single',[
            'analytic' => $analytic,
        ]);
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public static function create()
    {
        return Inertia::render('AdminDashboard/Analytics/Partials/Create');
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public static function edit(Request $request, $analytic)
    {
        Session::flash('url',$request->server('HTTP_REFERER'));
        return Inertia::render('AdminDashboard/Analytics/Partials/Update',[
            'analytic' => $analytic,
        ]);
    }

      /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public static function store(AnalyticStoreRequest $request)
    {

        if($request->hasFile('file')){
            $classifiedImg = $request->file('file');
            $filename = uniqid();
            Image::make($classifiedImg)->encode('webp')->save(public_path('storage/images/analytics/'  .  $filename . '.webp'));
            $path = 'images' .'/'. 'analytics' .'/'. $filename . '.webp';
        } else { $path = "none"; }

        $input = $request->validated();
        $input['thumbnail'] = $path;

        $res = $analytic = Analytic::create($input);

        if($res){
            session()->flash('flash.banner', 'Analytic Saved');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::route('admin-panel.analytics.index');
        }

        else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }
    }



     /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Models\Analytic  $Analytic
     * @return \Illuminate\Http\Response
     */
    public static function update(Request $request, $analytic)
    {


        if($request->hasFile('file')){
            Storage::disk('public')->delete($analytic->thumbnail);
            $classifiedImg = $request->file('file');
            $filename = uniqid();
            Image::make($classifiedImg)->encode('webp')->save(storage_path('app/public/images/analytics/'  .  $filename . '.webp'));
            $path = 'images' .'/'. 'analytics' .'/'. $filename . '.webp';
        } else if ($analytic->thumbnail != "none") { $path = $analytic->thumbnail; }
        else $path = "none";


        $input = $request->validated();
        $input['thumbnail'] = $path;

        $res =  $analytic->update($input);

        if($res){
            session()->flash('flash.banner', 'Analytic Saved');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::to(Session::get('url'));
        }

        else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }

    }


     /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Models\Analytic  $Analytic
     * @return \Illuminate\Http\Response
     */
    public static function destroy($analytic)
    {

        Storage::disk('public')->delete($analytic->thumbnail);


        $res = $analytic->delete();

        if($res){
            session()->flash('flash.banner', 'Analytic Removed');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::route('admin-panel.analytics.index');
        }

        else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }
    }

}

# Routing

Where possible, we use group or resource routes to keep our route files slim.

<?php

use Illuminate\Foundation\Application;

| Web Routes
|--------------------------------------------------------------------------
|
|
*/
