# Directory Structure

Laravel Docsopen in new window

Added Directory App\Http\Services

We use a very simplistic class extraction to move our business logic from the controller to a service class. We gave each model two service classes:

AdminServices - for handlers that are used in our Backend/CMS
WebServices - for handlers that are used on the public frontend and require no authorization.

These classes follow Laravel resource convention were possible, i.e. index, show,store etc. but will sometimes include custom functions.

# Validation

Laravel FormRequestopen in new windows

We validate every Store and Update Request with a FormRequest.
By default the authorize() function is set to true, because we use Policies in our Controllers to check the role and permissions of the user.

<?php

namespace App\Http\Requests;

use Illuminate\Foundation\Http\FormRequest;
use Str;

class PostRequest extends FormRequest
{
    /**
     * Determine if the user is authorized to make this request.
     *
     * @return bool
     */
    public function authorize()
    {
        return true;
    }

    /**
     * Get the validation rules that apply to the request.
     *
     * @return array
     */
    public function rules()
    {
        return [
            'title' => 'required|string',
            'category' => 'sometimes|in:none,one,two,three',
            'user_name' => 'sometimes|nullable',
            'status' => 'required|boolean',
            'content' => 'required',
            'images' => 'sometimes|nullable',
            'featured_image' => 'sometimes|nullable',
            'user' => 'required',
            'slug' => 'string'
        ];
    }

    protected function prepareForValidation()
    {
        // This is where the magic happens, the data from the key 'slug' will be overwritten
        // with the return value of the merge.

        $this->merge([
            'slug' => Str::slug($this->title)
        ]);
    }
}

# Controllers

Laravel Docsopen in new window

In our Controllers we use middleware to determine if a user has a certain role to ensure only those that are authorized to do so, can fire certain functions.

<?php

namespace App\Http\Controllers;

use App\Models\Post;
use Illuminate\Http\Request;

use App\Http\Services\Post\AdminServices;
use App\Http\Services\Post\WebServices;

use App\Http\Requests\PostRequest;

class PostController extends Controller
{
    public function __construct()
    {
        $this->authorizeResource(Post::class, 'post');
    }
    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index(Request $request)
    {
        return WebServices::index($request);
    }

    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function admin_index(Request $request)
    {
        $this->authorize('admin_index', Notification::class);
        return AdminServices::index($request);
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function create()
    {
        return WebServices::create();
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(PostRequest $request)
    {
        return AdminServices::store($request);
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public function show(Post $post)
    {
        return WebServices::show($post);
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function edit(Post $post)
    {
        return WebServices::edit($post);
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public function preview(Post $post)
    {
        $this->authorize('admin_index', Notification::class);
        return WebServices::preview($post);
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public function update(PostRequest $request, Post $post)
    {
        return AdminServices::update($request, $post);
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public function destroy(Post $post)
    {
        return AdminServices::destroy($post);
    }
}

# Services

We use service classes to put our logic in.
These services will always return an Inertia::Render or Redirect response.

Note we have to declare these functions static as our controllers call them statically.

<?php

namespace App\Http\Services\Post;

use App\Models\Post;
use Illuminate\Http\Request;
use Illuminate\Auth\Access\Response;
use Illuminate\Support\Facades\Storage;
use Illuminate\Filesystem\Filesystem;
use Illuminate\Support\Facades\File;
use Illuminate\Support\Facades\Validator;
use Illuminate\Support\Str;
use Image;

use App\Jobs\TestJob;

use Illuminate\Support\Facades\Redirect;
use Inertia\Inertia;

use App\Http\Requests\PostStoreRequest;

class AdminServices
{
    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public static function index(Request $request)
    {
        $ipp = $request->has('ipp') ? $request->ipp : 1000000;
        $orderBy = $request->has('orderBy') ? $request->orderBy : 'desc';
        $sort = $request->has('sort') ? $request->sort : 'created_at';

        return Inertia::render('CMS/Posts/Index', [
            'posts' => Post::orderBy($sort, $orderBy)
                ->when(!empty($request->get('search')), function ($query) use (
                    $request
                ) {
                    $query->where('title', 'like', "%{$request->search}%");
                })
                ->paginate(10)
                ->withQueryString()
                ->through(
                    fn($post) => [
                        'id' => $post->id,
                        'title' => $post->title,
                        'status' => $post->status
                    ]
                ),

            'filters' => [
                'search' => $request->search
            ]
        ]);
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public static function store($request)
    {
        $input = $request->validated();

        $user = auth('sanctum')->user();
        $input['user_id'] = $user->id;
        $input['user_name'] = $user->name;

        $res = Post::create($input);

        if ($res) {
            session()->flash('flash.banner', 'Post Saved');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::route('cms.posts');
        } else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Http\Response
     */
    public static function update($request, $id)
    {
        $input = $request->validated();

        $user = auth('sanctum')->user();
        $input['user_id'] = $user->id;
        $input['user_name'] = $user->name;

        $res = Post::find($id)->update($input);

        if ($res) {
            session()->flash('flash.banner', 'Post Saved');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::route('cms.posts');
        } else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Models\Event  $event
     * @return \Illuminate\Http\Response
     */
    public static function destroy(Post $post)
    {
        $images = $post->images;

        if (!empty($images)) {
            foreach ($images as $imageToDelete) {
                $fullPath = 'posts' . '/' . $imageToDelete;
                Storage::disk('public')->delete($fullPath);
            }
        }

        if ($post->featured_image) {
            Storage::disk('public')->delete(
                'posts' . '/' . $post->featured_image
            );
        }

        $res = $post->delete();

        if ($res) {
            session()->flash('flash.banner', 'Post Removed');
            session()->flash('flash.bannerStyle', 'success');
            return Redirect::route('cms.posts');
        } else {
            session()->flash('flash.banner', $res);
            session()->flash('flash.bannerStyle', 'danger');
            return Redirect::back();
        }
    }
}

# Policies

Laravel Docsopen in new window

We use a few policies that are tied to user roles to ensure a user is authorized to do something.

We use these policies in our Controllers.

Because we don't have a need for a very complex system, we've kept these policies rather small and simplistic. These policies will follow a cascading order from - always allowed to specifically allowed if user has a certain role.

<?php

namespace App\Policies;

use App\Models\Post;
use App\Models\User;
use Illuminate\Auth\Access\HandlesAuthorization;
use Illuminate\Auth\Access\Response;

class PostPolicy
{
    use HandlesAuthorization;

    public function before(User $user, $ability)
    {
        if ($user->isAdministrator || $user->isEditor) {
            return Response::allow();
        }
    }

    /**
     * Determine whether the user can view any models.
     *
     * @param  \App\Models\User  $user
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function admin_index(User $user)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can view any models.
     *
     * @param  \App\Models\User  $user
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function preview(User $user)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can view any models.
     *
     * @param  \App\Models\User  $user
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function viewAny(?User $user)
    {
        return Response::allow();
    }

    /**
     * Determine whether the user can view the model.
     *
     * @param  \App\Models\User  $user
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function view(?User $user, Post $post)
    {
        return Response::allow();
    }

    /**
     * Determine whether the user can create models.
     *
     * @param  \App\Models\User  $user
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function create(User $user)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can update the model.
     *
     * @param  \App\Models\User  $user
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function update(User $user, Post $post)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can delete the model.
     *
     * @param  \App\Models\User  $user
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function delete(User $user, Post $post)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can restore the model.
     *
     * @param  \App\Models\User  $user
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function restore(User $user, Post $post)
    {
        return Response::deny('You are not allowed to do this.');
    }

    /**
     * Determine whether the user can permanently delete the model.
     *
     * @param  \App\Models\User  $user
     * @param  \App\Models\Post  $post
     * @return \Illuminate\Auth\Access\Response|bool
     */
    public function forceDelete(User $user, Post $post)
    {
        return Response::deny('You are not allowed to do this.');
    }
}

# Observers

Laravel Docsopen in new window

Because a lot of our content is cached, we use Observers to make sure these caches are cleared when new content is added or content is updated.

<?php

namespace App\Observers;

use App\Models\Post;
use Illuminate\Support\Facades\Cache;

class PostObserver
{
    /**
     * Handle the Post "created" event.
     *
     * @param  \App\Models\Post  $post
     * @return void
     */
    public function created(Post $post)
    {
        Cache::flush(['home.latestnews', 'News_Index', 'News_Index_Latest']);
    }

    /**
     * Handle the Post "updated" event.
     *
     * @param  \App\Models\Post  $post
     * @return void
     */
    public function updated(Post $post)
    {
        Cache::flush(['home.latestnews', 'News_Index', 'News_Index_Latest']);
    }

    /**
     * Handle the Post "deleted" event.
     *
     * @param  \App\Models\Post  $post
     * @return void
     */
    public function deleted(Post $post)
    {
        Cache::flush(['home.latestnews', 'News_Index', 'News_Index_Latest']);
    }

    /**
     * Handle the Post "restored" event.
     *
     * @param  \App\Models\Post  $post
     * @return void
     */
    public function restored(Post $post)
    {
        Cache::flush(['home.latestnews', 'News_Index', 'News_Index_Latest']);
    }

    /**
     * Handle the Post "force deleted" event.
     *
     * @param  \App\Models\Post  $post
     * @return void
     */
    public function forceDeleted(Post $post)
    {
        Cache::flush(['home.latestnews', 'News_Index', 'News_Index_Latest']);
    }
}
